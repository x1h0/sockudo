## React

Use the built-in React entrypoint:

```ts
import Sockudo from "@sockudo/client";
import { SockudoProvider, useChannel } from "@sockudo/client/react";

export const client = new Sockudo("app-key", {
  cluster: "mt1",
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
});

function Feed() {
  const { subscribed, lastEventData } = useChannel("public-feed");
  return <pre>{JSON.stringify({ subscribed, lastEventData }, null, 2)}</pre>;
}

export function App() {
  return (
    <SockudoProvider client={client}>
      <Feed />
    </SockudoProvider>
  );
}
```

## Vue / Nuxt

Use the built-in Vue entrypoint:

```ts
import Sockudo from "@sockudo/client";
import { createSockudoPlugin, useChannel } from "@sockudo/client/vue";

const client = new Sockudo("app-key", {
  cluster: "mt1",
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
});

export default defineNuxtPlugin((nuxtApp) => {
  nuxtApp.vueApp.use(createSockudoPlugin(client));
});

const Feed = defineComponent({
  setup() {
    const { subscribed, lastEventData } = useChannel("public-feed");
    return () => h("pre", JSON.stringify({
      subscribed: subscribed.value,
      lastEventData: lastEventData.value,
    }, null, 2));
  },
});
```

## Web Worker

Use worker entrypoint:

```ts
import { SockudoWorker } from "@sockudo/client/worker";
```

## When to Stay with Echo/Pusher-js

If you only need core channel/event behavior and existing Echo ecosystem wrappers, keep them.  
Move to `@sockudo/client` when you need filters or delta compression.
