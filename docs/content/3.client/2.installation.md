## JavaScript / TypeScript

```bash
npm install @sockudo/client
```

Imports:

```ts
// Default (browser)
import { SockudoClient } from "@sockudo/client";

// Filter utilities
import { Filter } from "@sockudo/client/filter";

// Encrypted channels
import { SockudoClientEncrypted } from "@sockudo/client/with-encryption";

// Worker
import SockudoWorker from "@sockudo/client/worker";

// React hooks
import { SockudoProvider, useChannel } from "@sockudo/client/react";

// Vue composables
import { createSockudoPlugin, useChannel as useSockudoChannel } from "@sockudo/client/vue";
```

Framework peers:

```bash
npm install react react-dom vue
```

## Swift

Add the package to your Swift Package Manager manifest:

```swift
.package(url: "https://github.com/sockudo/sockudo-swift.git", from: "0.1.0")
```

```swift
import SockudoSwift
```

## Basic Connect

::code-group

```js [JavaScript]
import SockudoClient from "@sockudo/client";

const client = new SockudoClient("app-key", {
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
  // protocolVersion: 2 is the default
});

client.connect();
```

```swift [Swift]
let client = try SockudoClient(
    "app-key",
    options: .init(
        cluster: "local",
        forceTLS: false,
        enabledTransports: [.ws],
        wsHost: "127.0.0.1",
        wsPort: 6001,
        wssPort: 6001
    )
)
client.connect()
```

```kotlin [Kotlin]
val client = SockudoClient(
    "app-key",
    SockudoOptions(
        cluster = "local",
        forceTls = false,
        enabledTransports = listOf(SockudoTransport.ws),
        wsHost = "127.0.0.1",
        wsPort = 6001,
        wssPort = 6001,
    ),
)
client.connect()
```

```dart [Flutter]
final client = SockudoClient(
  'app-key',
  const SockudoOptions(
    cluster: 'local',
    forceTls: false,
    enabledTransports: <SockudoTransport>[SockudoTransport.ws],
    wsHost: '127.0.0.1',
    wsPort: 6001,
    wssPort: 6001,
  ),
);
client.connect();
```

::

## Disconnect

```js
client.disconnect();
```

## Runtime Requirements

- JavaScript: Node.js >= 22
- Swift: Xcode toolchain with Swift Package Manager
- Kotlin: Java 21+
- Flutter: Flutter 3.35+ and Dart 3.11+
