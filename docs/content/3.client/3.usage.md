## Subscribe and Bind

```ts
const channel = pusher.subscribe("public-updates");
channel.bind("price-updated", (data) => {
  console.log(data);
});
```

```swift
let channel = client.subscribe("public-updates")
channel.bind("price-updated") { data, _ in
    print(data ?? "")
}
```

```kotlin
val channel = client.subscribe("public-updates")
channel.bind("price-updated") { data, _ ->
    println(data)
}
```

```dart
final channel = client.subscribe('public-updates');
channel.bind('price-updated', (data, _) {
  print(data);
});
```

## Connect Lifecycle

```ts
pusher.connection.bind("connected", () => {
  console.log("connected", pusher.connection.socket_id);
});

pusher.connect();
```

```swift
client.connection.bind("connected") { data in
    print("connected", data ?? "")
}

client.connect()
```

```kotlin
client.connection.bind("connected") { data ->
    println("connected $data")
}

client.connect()
```

```dart
client.connection.bind('connected', (data) {
  print('connected $data');
});

client.connect();
```

## Private/Presence Auth

```ts
const pusher = new Pusher("app-key", {
  cluster: "mt1",
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
  channelAuthorization: {
    endpoint: "https://api.example.com/pusher/auth",
    transport: "fetch",
  },
});
```

```swift
let client = try SockudoClient(
    "app-key",
    options: .init(
        cluster: "local",
        forceTLS: false,
        wsHost: "127.0.0.1",
        wsPort: 6001,
        channelAuthorization: .init(
            endpoint: "https://api.example.com/pusher/auth"
        )
    )
)
```

```kotlin
val client =
    SockudoClient(
        "app-key",
        SockudoOptions(
            cluster = "local",
            forceTls = false,
            wsHost = "127.0.0.1",
            wsPort = 6001,
            channelAuthorization =
                ChannelAuthorizationOptions(
                    endpoint = "https://api.example.com/pusher/auth",
                ),
        ),
    )
```

```dart
final client = SockudoClient(
  'app-key',
  SockudoOptions(
    cluster: 'local',
    forceTls: false,
    wsHost: '127.0.0.1',
    wsPort: 6001,
    channelAuthorization: const ChannelAuthorizationOptions(
      endpoint: 'https://api.example.com/pusher/auth',
    ),
  ),
);
```

## User Signin

```ts
const pusher = new Pusher("app-key", {
  cluster: "mt1",
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
  userAuthentication: {
    endpoint: "https://api.example.com/pusher/user-auth",
    transport: "fetch",
  },
});

pusher.signin();
```

```swift
client.signIn()
```

```kotlin
client.signIn()
```

```dart
client.signIn();
```

## Filtered Subscriptions

```ts
import { Filter } from "@sockudo/client/filter";

const channel = pusher.subscribe("market-data", {
  filter: Filter.and(
    Filter.eq("event_type", "trade"),
    Filter.gte("price", "100")
  ),
});
```

```swift
let channel = client.subscribe(
    "market-data",
    options: .init(
        filter: .and([
            .eq("event_type", "trade"),
            .gte("price", "100")
        ])
    )
)
```

```kotlin
val channel =
    client.subscribe(
        "market-data",
        SubscriptionOptions(
            filter =
                Filter.and(
                    Filter.eq("event_type", "trade"),
                    Filter.gte("price", "100"),
                ),
        ),
    )
```

```dart
final channel = client.subscribe(
  'market-data',
  options: SubscriptionOptions(
    filter: Filter.and(<FilterNode>[
      Filter.eq('event_type', 'trade'),
      Filter.gte('price', '100'),
    ]),
  ),
);
```

## Delta Compression

```ts
const channel = pusher.subscribe("ticker:btc", {
  delta: { enabled: true, algorithm: "xdelta3" },
});
```

```swift
let channel = client.subscribe(
    "ticker:btc",
    options: .init(
        delta: .init(enabled: true, algorithm: .xdelta3)
    )
)
```

```kotlin
val channel =
    client.subscribe(
        "ticker:btc",
        SubscriptionOptions(
            delta = ChannelDeltaSettings(enabled = true, algorithm = DeltaAlgorithm.xdelta3),
        ),
    )
```

```dart
final channel = client.subscribe(
  'ticker:btc',
  options: const SubscriptionOptions(
    delta: ChannelDeltaSettings(
      enabled: true,
      algorithm: DeltaAlgorithm.xdelta3,
    ),
  ),
);
```

## Client Events

```ts
const channel = pusher.subscribe("private-chat");
channel.trigger("client-message", { text: "hello" });
```

```swift
let channel = client.subscribe("private-chat")
try channel.trigger("client-message", data: ["text": "hello"])
```

```kotlin
val channel = client.subscribe("private-chat")
channel.trigger("client-message", mapOf("text" to "hello"))
```

```dart
final channel = client.subscribe('private-chat');
channel.trigger('client-message', <String, Object>{'text': 'hello'});
```

Client event names must start with `client-`.

## Global Events

```ts
pusher.bind_global((eventName, data) => {
  console.log(eventName, data);
});
```

```swift
client.bindGlobal { eventName, data in
    print(eventName, data ?? "")
}
```

```kotlin
client.bindGlobal { eventName, data ->
    println("$eventName $data")
}
```

```dart
client.bindGlobal((eventName, data) {
  print('$eventName $data');
});
```

## Live Local Testing

All official clients are exercised against a local Sockudo instance on `127.0.0.1:6001`.

- Swift: `SOCKUDO_LIVE_TESTS=1 swift test`
- Kotlin: `SOCKUDO_LIVE_TESTS=1 ./gradlew :lib:test --tests io.sockudo.client.LiveIntegrationTest`
- Flutter: `SOCKUDO_LIVE_TESTS=1 flutter test`
