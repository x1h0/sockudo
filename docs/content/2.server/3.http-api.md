## Endpoints

### Realtime Publish

- `POST /apps/{appId}/events`
- `POST /apps/{appId}/batch_events`

### Channel Queries

- `GET /apps/{appId}/channels`
- `GET /apps/{appId}/channels/{channelName}`
- `GET /apps/{appId}/channels/{channelName}/users` (presence only)

### User Management

- `POST /apps/{appId}/users/{userId}/terminate_connections`

### Health and Ops

- `GET /up`
- `GET /up/{appId}`
- `GET /usage` (can be disabled via `http_api.usage_enabled: false`)
- `GET /metrics` (metrics server port, default `9601`)

## Signed Query Parameters

HTTP API routes under `/apps/*` require Pusher-style query auth:

- `auth_key`
- `auth_timestamp`
- `auth_version`
- `auth_signature`
- `body_md5` for non-empty `POST` body

Timestamp window is ±10 minutes.

## Signature Format

String to sign:

```text
{METHOD}\n{REQUEST_PATH}\n{SORTED_QUERY_EXCLUDING_AUTH_SIGNATURE}
```

Where query params are lowercase-key sorted pairs joined with `&`.

## Signing Example

::code-group

```ts [Node.js]
import crypto from "node:crypto";

export function signPusherRequest({
  method,
  path,
  appKey,
  appSecret,
  body,
}: {
  method: "GET" | "POST";
  path: string;
  appKey: string;
  appSecret: string;
  body?: string;
}) {
  const auth_timestamp = Math.floor(Date.now() / 1000).toString();
  const params: Record<string, string> = {
    auth_key: appKey,
    auth_timestamp,
    auth_version: "1.0",
  };

  if (method === "POST" && body && body.length > 0) {
    params.body_md5 = crypto.createHash("md5").update(body).digest("hex");
  }

  const queryForSig = Object.keys(params)
    .sort()
    .map((k) => `${k}=${params[k]}`)
    .join("&");

  const stringToSign = `${method}\n${path}\n${queryForSig}`;
  const auth_signature = crypto
    .createHmac("sha256", appSecret)
    .update(stringToSign)
    .digest("hex");

  return { ...params, auth_signature };
}
```

```go [Go]
package main

import (
    "crypto/hmac"
    "crypto/md5"
    "crypto/sha256"
    "encoding/hex"
    "fmt"
    "net/url"
    "sort"
    "strconv"
    "strings"
    "time"
)

func signRequest(method, path, appKey, appSecret, body string) url.Values {
    params := url.Values{}
    params.Set("auth_key", appKey)
    params.Set("auth_timestamp", strconv.FormatInt(time.Now().Unix(), 10))
    params.Set("auth_version", "1.0")

    if method == "POST" && body != "" {
        h := md5.Sum([]byte(body))
        params.Set("body_md5", hex.EncodeToString(h[:]))
    }

    keys := make([]string, 0, len(params))
    for k := range params {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    parts := make([]string, 0, len(keys))
    for _, k := range keys {
        parts = append(parts, k+"="+params.Get(k))
    }
    queryForSig := strings.Join(parts, "&")

    stringToSign := fmt.Sprintf("%s\n%s\n%s", method, path, queryForSig)
    mac := hmac.New(sha256.New, []byte(appSecret))
    mac.Write([]byte(stringToSign))
    params.Set("auth_signature", hex.EncodeToString(mac.Sum(nil)))
    return params
}
```

```php [PHP]
function signRequest(string $method, string $path, string $appKey, string $appSecret, string $body = ''): array {
    $params = [
        'auth_key'       => $appKey,
        'auth_timestamp' => time(),
        'auth_version'   => '1.0',
    ];

    if ($method === 'POST' && $body !== '') {
        $params['body_md5'] = md5($body);
    }

    ksort($params);
    $queryForSig = http_build_query($params);
    $stringToSign = implode("\n", [$method, $path, $queryForSig]);
    $params['auth_signature'] = hash_hmac('sha256', $stringToSign, $appSecret);

    return $params;
}
```

```python [Python]
import hashlib
import hmac
import time
from urllib.parse import urlencode

def sign_request(method: str, path: str, app_key: str, app_secret: str, body: str = "") -> dict:
    params = {
        "auth_key": app_key,
        "auth_timestamp": str(int(time.time())),
        "auth_version": "1.0",
    }

    if method == "POST" and body:
        params["body_md5"] = hashlib.md5(body.encode()).hexdigest()

    query_for_sig = urlencode(sorted(params.items()))
    string_to_sign = f"{method}\n{path}\n{query_for_sig}"
    signature = hmac.new(app_secret.encode(), string_to_sign.encode(), hashlib.sha256).hexdigest()
    params["auth_signature"] = signature
    return params
```

```ruby [Ruby]
require 'openssl'
require 'digest'

def sign_request(method, path, app_key, app_secret, body = '')
  params = {
    'auth_key'       => app_key,
    'auth_timestamp' => Time.now.to_i.to_s,
    'auth_version'   => '1.0',
  }

  params['body_md5'] = Digest::MD5.hexdigest(body) if method == 'POST' && !body.empty?

  query_for_sig = params.sort.map { |k, v| "#{k}=#{v}" }.join('&')
  string_to_sign = "#{method}\n#{path}\n#{query_for_sig}"
  auth_signature = OpenSSL::HMAC.hexdigest('sha256', app_secret, string_to_sign)

  params.merge('auth_signature' => auth_signature)
end
```

## Publish Example

::code-group

```ts [Node.js]
import { signPusherRequest } from "./sign";

const path = "/apps/app-id/events";
const body = JSON.stringify({
  name: "my-event",
  channel: "my-channel",
  data: JSON.stringify({ hello: "world" }),
});

const auth = signPusherRequest({
  method: "POST",
  path,
  appKey: "app-key",
  appSecret: "app-secret",
  body,
});

const qs = new URLSearchParams(auth).toString();
await fetch(`http://127.0.0.1:6001${path}?${qs}`, {
  method: "POST",
  headers: { "content-type": "application/json" },
  body,
});
```
